class Solution {
    public int findMaxLength(int[] nums) {
        
        int max = 0;
        
        for ( int i = 0; i < nums.length; i++){
            
            int zeroes = 0;
            int ones = 0;
            
            for (int j = i ; j < nums.length; j ++){
                

                if (nums[j] == 0){                    
                     zeroes++;                   
                }
                else{
                    ones++;
                }
                
                if(zeroes == ones){
                    max = Math.max(max,j - i +1);
                }
                   
                
            }
        }
        
        return max;
    }
}
